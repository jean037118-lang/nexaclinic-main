// ─── Tipos ────────────────────────────────────────────────────────────────────
type TipoContato = "ligacao" | "whatsapp" | "confirmacao" | "lembrete" | "retorno";
type StatusContato = "realizado" | "nao_atendeu" | "ocupado" | "cancelado" | "agendado";

interface HistoricoContato {
  id: string;
  patientName: string;
  phone: string;
  tipo: TipoContato;
  status: StatusContato;
  notas: string;
  data: string; // ISO
  profissionalId?: string;
  appointmentId?: string;
  appointmentDate?: string;
  appointmentStart?: string;
}

const HIST_KEY = "nexaclinic_telefonia_hist";

function loadHist(): HistoricoContato[] {
  try { return JSON.parse(localStorage.getItem(HIST_KEY) ?? "[]"); } catch { return []; }
}
function saveHist(h: HistoricoContato[]) {
  localStorage.setItem(HIST_KEY, JSON.stringify(h));
}
function addHist(entry: Omit<HistoricoContato, "id" | "data">) {
  const list = loadHist();
  const novo: HistoricoContato = { ...entry, id: `tel_${Date.now()}`, data: new Date().toISOString() };
  saveHist([novo, ...list]);
  return novo;
}

// ─── Helpers ──────────────────────────────────────────────────────────────────
function formatPhone(raw: string) {
  const d = raw.replace(/\D/g, "");
  if (d.length === 11) return `(${d.slice(0,2)}) ${d.slice(2,7)}-${d.slice(7)}`;
  if (d.length === 10) return `(${d.slice(0,2)}) ${d.slice(2,6)}-${d.slice(6)}`;
  return raw;
}
function phoneDigits(raw: string) { return raw.replace(/\D/g, ""); }

function whatsappURL(phone: string, msg: string) {
  const d = phoneDigits(phone);
  const num = d.startsWith("55") ? d : `55${d}`;
  return `https://wa.me/${num}?text=${encodeURIComponent(msg)}`;
}

function telURL(phone: string) {
  return `tel:+55${phoneDigits(phone)}`;
}

    window.open(telURL(phone), "_self");

                resultado:     { label: "Resultado de exame",       icon: Sparkles,        color: "border-rose-200 bg-rose-50 text-rose-700 hover:bg-rose-100" },
              }).map(([key, { label, icon: Icon, color }]) => (
                <button
                  key={key}
                  onClick={() => {
                    whatsapp(panelPaciente.name, panelPaciente.phone, key, panelPaciente.appointmentDate, panelPaciente.appointmentStart, panelPaciente.appointmentId);
                    setPanelOpen(false);
                  }}
                  className={`w-full flex items-center gap-3 rounded-xl border px-4 py-3 text-sm font-semibold transition ${color}`}
                >
                  <Icon className="h-4 w-4 shrink-0" />
                  {label}
                  <MessageCircle className="ml-auto h-3.5 w-3.5 opacity-50" />
                </button>
              ))}
            </div>

            {/* Registro manual */}
            <ManualRegistroPanel
              patientName={panelPaciente.name}
              phone={panelPaciente.phone}
              appointmentId={panelPaciente.appointmentId}
              onSave={(tipo, status, notas) => {
                registrar(panelPaciente.name, panelPaciente.phone, tipo, status, notas, panelPaciente.appointmentId, panelPaciente.appointmentDate, panelPaciente.appointmentStart);
                setPanelOpen(false);
                toast.success("Contato registrado no histórico");
              }}
            />
          </div>
        </div>
      )}
    </div>
  );
}

// ─── Subcomponente: registro manual ──────────────────────────────────────────
function ManualRegistroPanel({ patientName, phone, appointmentId, onSave }: {
  patientName: string;
  phone: string;
  appointmentId?: string;
  onSave: (tipo: TipoContato, status: StatusContato, notas: string) => void;
}) {
  const [open, setOpen] = useState(false);
  const [tipo, setTipo] = useState<TipoContato>("ligacao");
  const [status, setStatus] = useState<StatusContato>("realizado");
  const [notas, setNotas] = useState("");

  if (!open) {
    return (
      <div className="px-4 pb-4">
        <button
          onClick={() => setOpen(true)}
          className="w-full rounded-xl border border-dashed border-border px-4 py-2.5 text-xs font-semibold text-muted-foreground hover:border-primary/40 hover:text-foreground transition"
        >
          + Registrar contato manualmente
        </button>
      </div>
    );
  }

  return (
    <div className="border-t border-border px-4 py-4 space-y-3">
      <p className="text-xs font-semibold text-muted-foreground uppercase tracking-wide">Registrar contato</p>
      <div className="grid grid-cols-2 gap-2">
        <div>
          <p className="text-[10px] text-muted-foreground mb-1">Tipo</p>
          <select
            value={tipo}
            onChange={(e) => setTipo(e.target.value as TipoContato)}
            className="w-full border border-input rounded-xl px-2.5 py-1.5 text-xs bg-background focus:outline-none focus:ring-2 focus:ring-primary/40"
          >
            {Object.entries(tipoLabel).map(([k, v]) => <option key={k} value={k}>{v}</option>)}
          </select>
        </div>
        <div>
          <p className="text-[10px] text-muted-foreground mb-1">Status</p>
          <select
            value={status}
            onChange={(e) => setStatus(e.target.value as StatusContato)}
            className="w-full border border-input rounded-xl px-2.5 py-1.5 text-xs bg-background focus:outline-none focus:ring-2 focus:ring-primary/40"
          >
            {Object.entries(statusLabel).map(([k, v]) => <option key={k} value={k}>{v}</option>)}
          </select>
        </div>
      </div>
      <Input
        value={notas}
        onChange={(e) => setNotas(e.target.value)}
        placeholder="Observações (opcional)..."
        className="text-xs h-9"
      />
      <div className="flex gap-2">
        <Button size="sm" onClick={() => onSave(tipo, status, notas)} className="flex-1 gap-1.5 bg-primary text-primary-foreground">
          <Send className="h-3.5 w-3.5" /> Salvar registro
        </Button>
        <Button size="sm" variant="outline" onClick={() => setOpen(false)}>Cancelar</Button>
      </div>
    </div>
  );
}
